<?php
//Функция Шифровки сообщения message и установки шага(Shift)
function encrypt($message, $shift){
	$str = '';
	$array = str_split($message); // Разделение сообщения на однобайтовые элементы массива
	for($count = 0; $count<count($array);$count++){ // Цикл перебирающий элементы массива
		$lol = $array[$count]; // Переменная, куда загружаются элементы массива
		$str .= chr(ord($lol) + $shift); // перевод переменной lol в кодировку ASCII, сложение с шагом, далее перевод в стандартную кодировку, и добавление всех элементов lol к переменной str 
	}
	printf("%s : %d\n", $str,$shift);
	return $str;
}
//Функция Дешифровки, а так как дешифровка это шифровка со знаком минус, добавим знак минус к уже имеющейся функции
function decrypt($message, $shift){
	return encrypt($message,-$shift);
}
$message = "Hello"; // Поле ввода сообщения
$shift = 3; //Поле ввода шага
$encryptedMessage = encrypt($message, $shift); // Вызов функции шифрования
$decryptedMessage = decrypt($encryptedMessage, $shift); // Вызов функции дешифрования
?>
